ES6 - 解構賦值

陣列 - 解構賦值基本介紹

一般陣列取值寫法

let family = ['媽媽','爸爸','姊姊']; let mom = family[0]; let fa = family[1]; let sir = family[2]; console.log(mom , fa , sir);//媽媽 爸爸 姊姊

解構賦值的寫法

let family = ['媽媽','爸爸','姊姊']; let [mom , fa , sir] = family; console.log(mom , fa , sir);//媽媽 爸爸 姊姊

數量不對等時，會依序給予賦值

let family = ['媽媽','爸爸','姊姊']; let [mom , fa] = family; console.log(mom , fa);//媽媽 爸爸

如有空格，該值跳過賦值

let family = ['媽媽','爸爸','姊姊']; let [mom , , sir] = family; console.log(mom , sir);//媽媽 姊姊

陣列 - 變數應用

變數交換

let Pinky = '方廷'; let Erica = '霜霜'; [Erica,Pinky] = [Pinky,Erica]; console.log(Pinky,Erica);// 霜霜 方廷

拆解字元到變數上

let Team = '方廷和霜霜'; let [a , b , c , d , e]= Team; console.log(a , b , c , d , e);// 方 廷 和 霜 霜

預設值

let Fanny = '方芳'; let [Fanny2 = '預設F', Candy = '預設C'] = [Fanny]; console.log(Fanny2 , Candy); // 方芳 預設C

物件 - 解構賦值

物件解構賦值寫法

let OjectTeam = {  TeamOne : '喬巴',  TeamTwo : '魯夫',  TeamThree : '娜美' }; let { TeamOne , TeamTwo : NewTwo } = OjectTeam; // { 對應名稱 , 對應名稱 : 新名稱 } console.log(TeamOne , NewTwo);// 喬巴 魯夫

延伸問題

let OjectConetent = {  title : '雪寶',  content: ['阿尼','貓貓'] } let { title : NewTitle , content : [ , cat ] } = OjectConetent; console.log(NewTitle , cat);// 雪寶 貓貓

預設值

let { baby : NewBaby = '預設寶貝' } = {} console.log( NewBaby );// 預設寶貝
