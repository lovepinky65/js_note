建構式寫法

// 建構式內容 function Player(name,hp){  this.name = name;//屬性  this.hp = hp;//屬性  this.rest = function () {//方法   console.log(`${this.name}戰鬥中，被打了一下`);   this.hp--;  };  this.report = function(){//方法   console.log(`${this.name}的血量是${this.hp}`);  }; }
// player 建立第一個物件 var player = new Player('pinky','100'); player.report();
// player2 建立第二個物件 var player2 = new Player('Erica', '90'); player2.report(); player2.rest(); player2.report();
